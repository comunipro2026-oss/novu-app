create unique index if not exists idx_usuarios_telefono on novu_usuarios(telefono) where telefono is not null; create table if not exists novu_mensajes ( id uuid default gen_random_uuid() primary key, sala_id text not null, emisor_id text not null, emisor_nombre text, tipo text default 'texto', contenido text, created_at timestamptz default now() ); create index if not exists idx_mensajes_sala on novu_mensajes(sala_id, created_at); create table if not exists novu_contactos ( id uuid default gen_random_uuid() primary key, propietario_id text not null, contacto_telefono text not null, contacto_nombre text, created_at timestamptz default now(), unique(propietario_id, contacto_telefono) ); create index if not exists idx_contactos_propietario on novu_contactos(propietario_id); -- REALTIME alter table novu_mensajes replica identity full; alter table novu_usuarios replica identity full; do $$ begin if not exists (select 1 from pg_publication_tables where pubname = 'supabase_realtime' and tablename = 'novu_mensajes') then alter publication supabase_realtime add table novu_mensajes; end if; if not exists (select 1 from pg_publication_tables where pubname = 'supabase_realtime' and tablename = 'novu_usuarios') then alter publication supabase_realtime add table novu_usuarios; end if; end $$; -- RLS: acceso público (sin login) alter table novu_usuarios enable row level security; alter table novu_mensajes enable row level security; alter table novu_contactos enable row level security; create policy "novu_mensajes_all" on novu_mensajes for all using (true) with check (true); drop policy if exists "novu_contactos_all" on novu_contactos; create policy "novu_contactos_all" on novu_contactos for all using (true) with check (true); create policy "novu_media_read" on storage.objects for select using (bucket_id = 'novu-media'); drop policy if exists "novu_media_insert" on storage.objects; create policy "novu_media_insert" on storage.objects for insert with check (bucket_id = 'novu-media');
